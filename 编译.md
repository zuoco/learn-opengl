cmake -S . \
  -B build \
  -GNinja \
  -DCMAKE_C_COMPILER=clang \
  -DCMAKE_CXX_COMPILER=clang++ \
  -DCMAKE_BUILD_TYPE=Release \
  -DBUILD_SHARED_LIBS=OFF \
  -DCMAKE_INSTALL_PREFIX=install \
  -DSLINT_FEATURE_BACKEND_WINIT=ON \
  -DSLINT_FEATURE_BACKEND_QT=OFF \
  -DSLINT_FEATURE_BACKEND_LINUXKMS=ON \
  -DSLINT_FEATURE_RENDERER_FEMTOVG=ON \
  -DSLINT_FEATURE_RENDERER_SKIA=ON \
  -DSLINT_FEATURE_RENDERER_SOFTWARE=ON \
  -DSLINT_FEATURE_INTERPRETER=OFF \
  -DSLINT_FEATURE_TESTING=OFF \
  -DSLINT_FEATURE_EXPERIMENTAL=OFF

# 多核编译
cmake --build build -j$(nproc)
# 或者
cmake --build build --parallel

# 编译完成后安装
cmake --install build
